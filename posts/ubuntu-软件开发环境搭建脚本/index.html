#!/bin/bash

PATH=/bin:/sbin:/usr/bin:/usr/sbin:/usr/local/bin:/usr/local/sbin:~/bin
export PATH

clear
# 脚本信息
installer_ver=1.0.3
installer_info="请联系作者，QQ：422803653"

# 定义颜色动作
SETCOLOR_SUCCESS="echo -en \\E[1;32m"
SETCOLOR_FAILURE="echo -en \\E[1;31m"
SETCOLOR_WARNING="echo -en \\E[1;33m"
SETCOLOR_NORMAL="echo -en  \\E[0;39m"

# 使用时直接调用颜色动作,跟上相应的内容
# $SETCOLOR_SUCCESS && echo test1
# $SETCOLOR_FAILURE && echo test2
# $SETCOLOR_WARNING && echo test3
# $SETCOLOR_NORMAL  && echo test4

#jetbrain 软件参数
jetbrain_url_prefix="https://download.jetbrains.8686c.com/"
jetbrains_version="2020.1"

init_apt_source() {
	# 备份
	sudo cp -a /etc/apt/sources.list /etc/apt/sources.list.bak
	# 修改
	sudo sed -i "s@http://.*archive.ubuntu.com@http://mirrors.aliyun.com@g" /etc/apt/sources.list
	sudo sed -i "s@http://.*security.ubuntu.com@http://mirrors.aliyun.com@g" /etc/apt/sources.list
	# 更新
	sudo apt update

	$SETCOLOR_SUCCESS && echo apt阿里源设置成功
}

init_pip_source() {

	mkdir ~/.pip

	touch ~/.pip/pip.conf

	sed -i "[global]" ~/.pip/pip.conf
	sed -i "trusted-host=mirrors.aliyun.com" ~/.pip/pip.conf
	sed -i "index-url=https://mirrors.aliyun.com/pypi/simple/" ~/.pip/pip.conf

	$SETCOLOR_SUCCESS && echo pip阿里源设置成功

}

# 安装常用软件
init_installer() {
	sudo apt-get update
	sudo apt-get -y install openssh-server curl wget
}

#docker 安装
docker_ce_installer() {
	# 检查docker是否已安装
	sudo docker version &>/dev/null
	if [ $? -eq 0 ]; then
		$SETCOLOR_WARNING && echo 您已经安装了docker,不用再次安装
		$SETCOLOR_NORMAL && echo Bye ^_^
		exit 1
	else
		# step 1: 安装必要的一些系统工具
		sudo apt-get update
		sudo apt-get -y install apt-transport-https ca-certificates software-properties-common
		# step 2: 安装GPG证书
		curl -fsSL https://mirrors.aliyun.com/docker-ce/linux/ubuntu/gpg | sudo apt-key add -

		# Step 3: 写入软件源信息
		sudo add-apt-repository "deb [arch=amd64] https://mirrors.aliyun.com/docker-ce/linux/ubuntu $(lsb_release -cs) stable"
		# Step 4: 更新并安装Docker-CE
		sudo apt-get -y update
		sudo apt-get -y install docker-ce
		# Step 5: 将当前用户加入到docker用户组中
		sudo usermod -aG docker $USER
		# Step 6: 安装docker-compose
		sudo curl -L "https://github.com/docker/compose/releases/download/1.25.5/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
		sudo chmod +x /usr/local/bin/docker-compose
		sudo ln -s /usr/local/bin/docker-compose /usr/bin/docker-compose

		$SETCOLOR_SUCCESS && echo docker 安装成功
	fi
}

goland_installer() {
	# goland url信息
	goland_url="$jetbrain_url_prefix/python/pycharm-professional-$jetbrains_version.tar.gz"
	wget $goland_url
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 获取安装包失败，${installer_info}

		exit 1

	else
		golandsrc=$(ls | grep goland)
		sudo tar -C /opt/jetbrain/ -zxvf $golandsrc
		rm $golandsrc

		$SETCOLOR_SUCCESS && echo goland安装成功
	fi

}

pycharm_installer() {
	# pycharm url信息
	pycharm_url="$jetbrain_url_prefix/go/goland-$jetbrains_version.tar.gz"
	wget $pycharm_url
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 获取安装包失败，${installer_info}

		exit 1

	else
		pycharmsrc=$(ls | grep pycharm)
		sudo tar -C /opt/jetbrain/ -zxvf $pycharmsrc
		rm $pycharmsrc
		$SETCOLOR_SUCCESS && echo pycharm 安装成功
	fi

}

# v2ray linux GUI client
v2rayA_installer() {
	latest_url="https://github.com/mzz2017/v2rayA/releases/latest"
	latest_version=$(curl -s https://github.com/mzz2017/v2rayA/releases/latest | awk 'BEGIN{FS="\""}{print $2}' | awk 'BEGIN{FS="/"}{print $8}')
	v2rayA_url="https://github.com/mzz2017/v2rayA/releases/download/$latest_version/installer_debian_amd64_$latest_version.deb"
	wget $v2rayA_url
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 获取安装包失败，${installer_info}

		exit 1

	else
		sudo dpkg -i installer_debian_amd64_$latest_version.deb
		rm installer_debian_amd64_$latest_version.deb
		$SETCOLOR_SUCCESS && echo v2rayA 安装成功
	fi

}
# miniconda
miniconda_installer() {
	miniconda_url="https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh"
	wget $miniconda_url
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 获取安装包失败，${installer_info}

		exit 1

	else
		chmod a+x Miniconda3-latest-Linux-x86_64.sh
		./Miniconda3-latest-Linux-x86_64.sh
		rm Miniconda3-latest-Linux-x86_64.sh
		$SETCOLOR_SUCCESS && echo miniconda 安装成功
	fi

}
# golang
golang_installer() {

	go version &>/dev/null
	if [ $? -eq 0 ]; then
		$SETCOLOR_WARNING && echo 您已经安装了go,不用再次安装
		$SETCOLOR_NORMAL && echo Bye ^_^ \n
		break
	fi

	$SETCOLOR_NORMAL && echo 当前环境允许安装，你确认要开始安装吗[y | N]:
	read -n1 install_choose
	echo -e "\n"
	if [[ $install_choose == 'y' ]] || [[ $install_choose == 'Y' ]]; then
		$SETCOLOR_NORMAL && echo OK, 请稍后，马上就好.

	else
		$SETCOLOR_SUCCESS && echo 取消成功.
		exit 0
	fi

	#下载最新的go版本
	gourl=$(curl -s https://studygolang.com/dl | sed -n '/dl\/golang\/go.*\.linux-amd64\.tar\.gz/p' | sed -n '1p' | sed -n '/1/p' | awk 'BEGIN{FS="\""}{print $4}')
	goweb="https://studygolang.com"
	gourl="${goweb}${gourl}"
	#防止已经下载过
	if [ ! -f "$(ls | grep linux-amd64.tar.gz | sed -n '1p')" ]; then
		wget $gourl
		if [ $? -ne 0 ]; then
			$SETCOLOR_WARNING && echo 获取安装包失败，${give_info}
			exit 1
		fi
	else
		$SETCOLOR_WARNING && echo 文件已存在.

	fi

	gosrc=$(ls | grep linux-amd64.tar.gz | sed -n '1p')
	#下载完成后解压到对应的目录
	installPath="/usr/local"
	if [[ -f "$gosrc" ]] && [[ -d "$installPath" ]] && [[ ! -d "$installPath/go" ]]; then
		$SETCOLOR_NORMAL && echo 开始解压，${give_info}
		sudo tar -C /usr/local -zxvf $gosrc
		if [ $? -ne 0 ]; then
			$SETCOLOR_FAILURE && echo 解压失败，${give_info}

			exit 1
		else
			rm $gosrc
		fi
	else
		$SETCOLOR_FAILURE && echo 安装失败，${give_info}
	fi
	# 导入环境变量
	pathFile="/etc/profile"
	if [ ! -f "$pathFile" ]; then
		$SETCOLOR_FAILURE && echo $pathFile 文件不存在
		exit 1
	fi

	# 导入之前应该先判断是否已经设置过对应的值[不仅仅是etc/profile这一个配置文件] 如果设置过  提示 1-覆盖 2-跳过并手动添加
	sudo echo 'export GOROOT=/usr/local/go' >>$pathFile
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 导入环境变量失败，${give_info}

		exit 1
	fi
	sudo echo 'export PATH=$PATH:$GOROOT/bin' >>$pathFile
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 导入环境变量失败，${give_info}

		exit 1
	fi

	source $pathFile
	if [ $? -ne 0 ]; then
		$SETCOLOR_FAILURE && echo 导入环境变量失败，${give_info}

		exit 1
	fi

	#再次验证安装
	go version
	if [ $? -eq 0 ]; then
		$SETCOLOR_SUCCESS && echo 您已经成功安装了go

	else
		$SETCOLOR_FAILURE && echo 安装失败

	fi
}

# terminal
show_options() {
	printf "
#######################################################################
#       软件环境 V${installer_ver} for ubuntu, Written by laoxie       
#######################################################################
#       1-修改为阿里软件源;                                             
#       2-安装docker;                         
#       3-安装golang;                                               
#       4-安装miniconda;                                    
#       5-安装科学上网工具;                                     
#       6-安装pycharm;                                           
#       7-安装goland;
#		8-安装typora;                                    
#       e-exit;                                                       
#######################################################################
"
	echo -e -n "\033[01;36m请输入您的选择: \033[0m"
}

main() {
	while
		show_options
		read -p "" choose
	do
		source /etc/profile &>/dev/null
		echo -e "\n"
		case $choose in
		1)
			init_apt_source

			init_pip_source

			init_installer

			;;
		2)
			docker_ce_installer
			;;
		3)

			golang_installer
			;;
		4)

			miniconda_installer
			;;
		5)
			v2rayA_installer
			;;
		6)
			pycharm_installer
			;;

		7)
			goland_installer
			;;
		8)
			typora_installer
			;;
		e | E)
			$SETCOLOR_WARNING && echo Bye 亲 .
			exit 0
			;;
		*)
			$SETCOLOR_WARNING && echo 亲，别逗我，请按照提示输入.
			;;
		esac
	done

}
